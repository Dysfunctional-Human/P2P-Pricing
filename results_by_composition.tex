\section*{Table 3: Decision Guide}

\begin{table}[h!]
\centering
\renewcommand{\arraystretch}{1.6}
\begin{tabular}{L{2.2cm} L{4.2cm} L{3.8cm} L{5cm}}
\toprule
\textbf{Mechanism} & \textbf{Best Suited For} & \textbf{Community Profile} & \textbf{Key Trade-off} \\
\midrule
\cellcolor{wingreen}\textbf{SDR + DSM} &
  Any community where maximum cost reduction is the primary goal &
  Consumer-heavy (limited PV but flexible loads) &
  Highest savings by far; prosumers gain 3--8$\times$ more than consumers ($\mathcal{F}$ = 0.24--0.53) \\[6pt]
\cellcolor{winblue}\textbf{MMR} &
  Balanced or diverse communities seeking fair cost sharing &
  20--50\% consumers; mixed PV sizes &
  Nearly equal group savings (Con $\approx$ Pro); $\mathcal{F}$ = 0.85--1.0; robust across compositions \\[6pt]
\cellcolor{winorange}\textbf{Bill-Sharing} &
  Prosumer-dominated communities needing guaranteed equity &
  $C \leq 15\%$; many large PV systems &
  Consumer and prosumer savings nearly identical ($\mathcal{F} > 0.93$); never harms prosumers \\
\bottomrule
\end{tabular}
\caption*{No single mechanism dominates. SDR maximizes total savings but creates inequity. MMR balances savings and fairness. Bill-Sharing guarantees equity at the cost of lower total savings. The optimal choice is a \emph{policy} decision driven by community values.}
\end{table}
